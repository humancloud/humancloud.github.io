微软面试开发工程师最常用的问题: 把一个字符串转成整数


//这题与那道  将字符串中的数字提取出来组成数字不一样,这个是把整个字符串转为数字
int StrtoInt(char *string)
{
    int num = 0;
    while(*string != 0)
    {
        num = num*10 + *string - '\0';
        ++string;
    }
    return num;
}


'a' - '0' 是他的ascii码




//正确做法
int StrtoInt(char *string)
{
    if(string == NULL){
        return -1;
    }
    int num = 0;
    while(*string != 0)
    {
        num = num*10 + *string - '\0';
        ++string;
    }
    return num;
}
